class Solution {
    public int minSubArrayLen(int s, int[] nums) {

        int len = nums.length;
        int windowlen = Integer.MAX_VALUE;
        int i = 0;
        int sum = 0;
        for (int j = 0; j < len; ++j) {
            sum += nums[j];
            while (sum >= s) {
                windowlen = Math.min (windowlen, j - i + 1);
                sum -= nums[i];
                i++;
            }
        }
        return windowlen == Integer.MAX_VALUE ? 0 : windowlen;

    }
}

        leng = len(nums)
        windowlen = sys.maxsize
        i = 0
        sum = 0
        for j in range(0, leng):

            while sum >= s:
                windowlen = min(windowlen, j - i + 1)
                sum -= nums[i]
                i += 1

        if windowlen == sys.maxsize:
            return 0
        else:
            return windowlen

func minSubArrayLen(target int, nums []int) int {
    length := len(nums)
    winLen := length + 1
    i := 0
    sum := 0
    for j := 0; j < length; j++ {
        sum += nums[j]
        for sum >= target {
            winLen = min(winLen, j - i + 1)
            sum -= nums[i]
            i++
        }
    }
    if winLen == length + 1 {
        return 0
    }
    return winLen
}

func min(a, b int) int {
    if a < b {
        return a
    }
    return b
}
