class Solution {
    public int removeElement(int[] nums, int val) {
        //è·å–æ•°ç»„é•¿åº¦
        int len = nums.length;
        if (len == 0) {
            return 0;
        }
        int i = 0;
        for (i = 0; i < len; ++i) {
            //å‘ç°ç¬¦åˆæ¡ä»¶çš„æƒ…å†?
            if (nums[i] == val) {
                //å‰ç§»ä¸€ä½?

                    nums[j] = nums[j+1];

                i--;
                len--;
            }
        }
        return i;
    }
}
