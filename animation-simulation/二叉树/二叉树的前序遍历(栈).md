func preorderTraversal(root *TreeNode) []int {
    res := []int{}
    if root == nil {
        return res
    }
    stk := []*TreeNode{root}
    for len(stk) != 0 {
        temp := stk[len(stk) - 1]
        stk = stk[: len(stk) - 1]
        if temp.Right != nil {
            stk = append(stk, temp.Right)
        }
        if temp.Left != nil {
            stk = append(stk, temp.Left)
        }
        res = append(res, temp.Val)
    }
    return res
}
